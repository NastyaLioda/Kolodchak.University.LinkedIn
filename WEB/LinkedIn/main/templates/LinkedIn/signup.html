from django.contrib.auth.forms import UserCreationForm from django.urls import reverse_lazy from django.shortcuts import render, redirect def signup(request): if request.method == 'POST': form = UserCreationForm(request.POST) if form.is_valid(): form.save() # Після успішної реєстрації перенаправити користувача на бажану сторінку return redirect(reverse_lazy('home')) # Використовуйте reverse_lazy для отримання URL else: form = UserCreationForm() return render(request, 'signup.html', {'form': form})
